create table ARTICLE (
	ARTICLE_ID INTEGER not null,
	USER_ID TEXT,
	TITLE TEXT,
	CONTENT TEXT,
	URL_TILE TEXT,
	COVER_IMAGE_URL TEXT,
	CREATION_DATETIME DATETIME,
	LAST_MODIFIED_DATETIME DATETIME,
	STATUS INTEGER,
	primary key (ARTICLE_ID)
);
create table TAG (
	TAG_ID INTEGER not null,
	DISPLAY_NAME TEXT,
	USED_COUNT INTEGER,
	LAST_USED_DATETIME DATETIME,
	primary key (TAG_ID)
);
create table ARTICLE_TAG_EDGE (
	TAG_ID INTEGER not null,
	ARTICLE_ID INTEGER not null,
	primary key (TAG_ID, ARTICLE_ID),
	constraint FK_ARTICLE_TAG_EDGE_TAG foreign key (TAG_ID) references TAG,
	constraint FK_ARTICLE_TAG_EDGE_ARTICLE foreign key (ARTICLE_ID) references ARTICLE
);
create table HIT (
	HIT_ID INTEGER not null,
	RESOURCE_ID INTEGER,
	IP_ADDRESS TEXT,
	HIT_TYPE INTEGER,
	CREATION_DATETIME DATETIME,
	primary key (HIT_ID)
);
create table SIMPLE_USER (
	USER_ID TEXT not null,
	LOGIN_ID TEXT,
	PASSWORD TEXT,
	DISPLAY_NAME TEXT,
	primary key (USER_ID)
);

create index "IX_FK_ARTICLE_TAG_EDGE_1" on ARTICLE_TAG_EDGE ("TAG_ID" ASC);
create index "IX_FK_ARTICLE_TAG_EDGE_2" on ARTICLE_TAG_EDGE ("ARTICLE_ID" ASC);
